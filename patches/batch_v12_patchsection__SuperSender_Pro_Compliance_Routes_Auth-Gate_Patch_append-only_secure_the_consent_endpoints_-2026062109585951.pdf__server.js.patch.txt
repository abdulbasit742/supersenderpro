Find where the compliance module is registered (e.g. compliance.register(app) or
require('./src/modules/compliance').register(app) ). Insert this block immediately before that line. If

not, this block attaches it scoped to compliance.


  // BEGIN COMPLIANCE AUTH GATE HOOK
  try {
    const complianceRouteGuard = require('./lib/adminAuth/complianceRouteGuard');
      // Ensure req.adminAuth is populated for the compliance paths (no-op if already attached app-wide).
      if (complianceRouteGuard.attach) {
       app.use('/api/compliance', complianceRouteGuard.attach());
       app.use('/compliance', complianceRouteGuard.attach());
      }
      // Gate sensitive compliance API operations (mutations + PII reads); leaves /inbound + /status open.
      app.use('/api/compliance', complianceRouteGuard.apiGuard());
      // Gate the /compliance dashboard page (toggle via COMPLIANCE_GATE_DASHBOARD).
    app.use('/compliance', complianceRouteGuard.dashboardGuard());
  } catch (e) {
      console.error('[complianceAuthGate] not mounted:', e && e.message);
  }
  // END COMPLIANCE AUTH GATE HOOK
